// Helper to generate "fake" history that ends exactly at the current live price
const generateSeamlessHistory = (endPrice, count, volatility = 0.0001) => {
  let data = [];
  let currentPrice = endPrice;
  
  for (let i = 0; i < count; i++) {
    const close = currentPrice;
    // Lower volatility for Forex compared to Crypto
    const open = close - (Math.random() - 0.5) * close * volatility * 0.5;
    const high = Math.max(open, close) + Math.random() * close * volatility * 0.3;
    const low = Math.min(open, close) - Math.random() * close * volatility * 0.3;
    
    data.unshift({ time: i, open, high, low, close });
    currentPrice = open;
  }
  return data;
};

const Chart = ({ data, activeSymbolObj, currentPrice, timeframe }) => {
  const containerRef = useRef(null);
  const [dimensions, setDimensions] = useState({ width: 800, height: 400 });

  useEffect(() => {
    const handleResize = () => {
      if (containerRef.current) {
        setDimensions({
          width: containerRef.current.offsetWidth,
          height: containerRef.current.offsetHeight
        });
      }
    };
    handleResize();
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  if (!data || data.length === 0) return <div className="flex-1 bg-black flex items-center justify-center text-gray-600 text-xs">Waiting for tick...</div>;

  const displayDecimals = activeSymbolObj?.digits || 4;

  const minPrice = Math.min(...data.map(d => d.low));
  const maxPrice = Math.max(...data.map(d => d.high));
  const priceRange = maxPrice - minPrice || 1;
  
  const pY = priceRange * 0.1; // 10% padding
  const minY = minPrice - pY;
  const maxY = maxPrice + pY;
  const rangeY = maxY - minY;

  const candleWidth = dimensions.width / (data.length + 15); 
  const gap = candleWidth * 0.3;

  const getY = (price) => dimensions.height - ((price - minY) / rangeY) * dimensions.height;
  const getX = (index) => index * candleWidth + 10;

  const currentY = getY(currentPrice);

  return (
    <div className="flex-1 bg-black relative overflow-hidden cursor-crosshair" ref={containerRef}>
        <div className="absolute inset-0 flex items-center justify-center pointer-events-none opacity-10">
            <span className="text-6xl font-bold text-gray-500">{activeSymbolObj.display}</span>
        </div>

      <svg width="100%" height="100%" className="absolute inset-0">
        {/* Grid Lines - Horizontal */}
        {[0, 0.2, 0.4, 0.6, 0.8, 1].map(pct => {
            const y = dimensions.height * pct;
            const price = maxY - (pct * rangeY);
            return (
                <g key={pct}>
                    <line x1="0" y1={y} x2="100%" y2={y} stroke="#333" strokeDasharray="4 4" strokeWidth="0.5" />
                    <text x={dimensions.width - 5} y={y - 5} textAnchor="end" fill="#666" fontSize="10">
                        {price.toFixed(displayDecimals)}
                    </text>
                </g>
            )
        })}

        {/* Candles */}
        {data.map((d, i) => {
            const x = getX(i);
            const yOpen = getY(d.open);
            const yClose = getY(d.close);
            const yHigh = getY(d.high);
            const yLow = getY(d.low);
            const isGreen = d.close >= d.open;
            const color = isGreen ? '#26a69a' : '#ef5350';
            const bodyTop = Math.min(yOpen, yClose);
            const bodyHeight = Math.abs(yClose - yOpen) || 1;

            return (
                <g key={i}>
                    <line x1={x + candleWidth/2} y1={yHigh} x2={x + candleWidth/2} y2={yLow} stroke={color} strokeWidth="1" />
                    <rect x={x + gap} y={bodyTop} width={candleWidth - gap*2} height={bodyHeight} fill={color} stroke={color} />
                </g>
            );
        })}

        {/* Current Price Line */}
        <line x1="0" y1={currentY} x2="100%" y2={currentY} stroke="#3b82f6" strokeWidth="1" />
        <rect x={dimensions.width - 70} y={currentY - 10} width="70" height="20" fill="#3b82f6" />
        <text x={dimensions.width - 35} y={currentY + 4} textAnchor="middle" fill="white" fontSize="11" fontWeight="bold">
            {currentPrice.toFixed(displayDecimals)}
        </text>
      </svg>

      {/* Chart Info Overlay */}
      <div className="absolute top-2 left-2 text-gray-400 text-xs font-mono bg-black/50 p-1 rounded z-10">
        <span className="text-white font-bold mr-2">{activeSymbolObj.display}, {timeframe}</span>
        O: <span className="text-yellow-500">{data[data.length-1].open.toFixed(displayDecimals)}</span>{' '}
        H: <span className="text-green-500">{data[data.length-1].high.toFixed(displayDecimals)}</span>{' '}
        L: <span className="text-red-500">{data[data.length-1].low.toFixed(displayDecimals)}</span>{' '}
        C: <span className="text-blue-400">{data[data.length-1].close.toFixed(displayDecimals)}</span>
      </div>
    </div>
  );
};

const OrderPanel = ({ symbolObj, price, onTrade }) => {

    const rawBid = parseFloat(price.bid);
    const rawAsk = parseFloat(price.ask);
    // If rawAsk == rawBid, add simulated spread
    const ask = rawAsk > rawBid ? rawAsk : rawBid + (symbolObj.spread * (symbolObj.name.includes('USDT') ? 1 : 0.0001));
    const bid = rawBid;

    return (
        <div className="absolute top-12 right-12 bg-[#1e222d] border border-gray-600 rounded shadow-xl p-2 w-48 z-10 opacity-90 hover:opacity-100 transition-opacity">
            <div className="flex justify-between items-center mb-2 text-xs text-gray-300">

export default function App() {
  const [activeSymbol, setActiveSymbol] = useState('EURUSDT');
  const [timeframe, setTimeframe] = useState('M1');
  const [prices, setPrices] = useState({});
  const [chartData, setChartData] = useState([]);
  const [positions, setPositions] = useState([]);
  const [balance, setBalance] = useState(INITIAL_BALANCE);
  const [connectionStatus, setConnectionStatus] = useState('Connecting');
  const wsRef = useRef(null);

  // Get active symbol object
  const activeSymbolObj = SYMBOLS.find(s => s.name === activeSymbol) || SYMBOLS[0];

  // WebSocket Connection
  useEffect(() => {
    const streams = SYMBOLS.map(s => `${s.name.toLowerCase()}@ticker`).join('/');
    const wsUrl = `wss://stream.binance.com:9443/ws/${streams}`;

    setConnectionStatus('Connecting');
    const ws = new WebSocket(wsUrl);
    wsRef.current = ws;

    ws.onopen = () => {
      setConnectionStatus('Connected');
    };

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.s && data.b && data.a) {
        setPrices(prev => {
          const prevPrice = prev[data.s]?.bid || 0;
          const newBid = parseFloat(data.b);
          
          return {
            ...prev,
            [data.s]: {
              bid: data.b,
              ask: data.a,
              close: parseFloat(data.c),
              direction: newBid >= prevPrice ? 'up' : 'down'
            }
          };
        });
      }
    };

    ws.onclose = () => setConnectionStatus('Disconnected');
    ws.onerror = () => setConnectionStatus('Error');

    return () => {
      if (ws.readyState === 1) ws.close();
    };
  }, []);

  // Chart Data Management (Hybrid approach)
  useEffect(() => {
    setChartData([]);
  }, [activeSymbol]);

  // Update Chart with Live Price
  useEffect(() => {
    const priceInfo = prices[activeSymbol];
    if (!priceInfo) return;

    const currentPrice = priceInfo.close;

    setChartData(prev => {
      // 1. If no data, generate "history" ending at this price
      if (prev.length === 0) {
        return generateSeamlessHistory(currentPrice, 80, 0.00005); // Lower volatility init for forex
      }

      // 2. Update last candle
      const lastCandle = prev[prev.length - 1];
      const newCandle = {
        ...lastCandle,
        close: currentPrice,
        high: Math.max(lastCandle.high, currentPrice),
        low: Math.min(lastCandle.low, currentPrice)
      };

      // 3. Logic to start new candle 
      if (Math.random() > 0.99) { // Slower candles for forex stability
         return [...prev, {
           time: prev.length,
           open: currentPrice,
           close: currentPrice,
           high: currentPrice,
           low: currentPrice
         }].slice(-100);
      } else {
         return [...prev.slice(0, -1), newCandle];
      }
    });

  }, [prices, activeSymbol]);

  // Trade Logic
  const handleTrade = (type, volume, price) => {
    const newPosition = {
        id: Math.floor(Math.random() * 10000000),
        time: new Date().toLocaleTimeString(),
        type,
        volume,
        symbolName: activeSymbol,
        displaySymbol: activeSymbolObj.display,
        openPrice: price,
        digits: activeSymbolObj.digits
    };
    setPositions([...positions, newPosition]);
  };

  const closePosition = (id) => {
      const pos = positions.find(p => p.id === id);
      if(!pos) return;

      const priceData = prices[pos.symbolName];
      const currentPrice = priceData 
        ? (pos.type === 'buy' ? parseFloat(priceData.bid) : parseFloat(priceData.ask))
        : pos.openPrice;

      const diff = pos.type === 'buy' ? currentPrice - pos.openPrice : pos.openPrice - currentPrice;
      const multiplier = pos.symbolName.includes('PAXG') ? 1 : 10000;
      const profit = diff * pos.volume * multiplier;

      setBalance(prev => prev + profit);
      setPositions(prev => prev.filter(p => p.id !== id));
  };

  // Equity Calculation
  const equity = useMemo(() => {
    let runningProfit = 0;
    positions.forEach(pos => {
        const priceData = prices[pos.symbolName];
        if (!priceData) return;
        
        const currentPrice = pos.type === 'buy' ? parseFloat(priceData.bid) : parseFloat(priceData.ask);
        const diff = pos.type === 'buy' ? currentPrice - pos.openPrice : pos.openPrice - currentPrice;
        const multiplier = pos.symbolName.includes('PAXG') ? 1 : 10000;
        runningProfit += diff * pos.volume * multiplier;
    });
    return balance + runningProfit;
  }, [balance, positions, prices]);

  return (
    <div className="flex flex-col h-screen bg-[#131722] text-gray-300 overflow-hidden font-sans">
      <Header connectionStatus={connectionStatus} />
      <Toolbar currentTimeframe={timeframe} setTimeframe={setTimeframe} />
      
      <div className="flex flex-1 overflow-hidden">
        <MarketWatch 
            symbols={SYMBOLS} 
            prices={prices} 
            activeSymbol={activeSymbol} 
            onSelectSymbol={setActiveSymbol} 
        />
        
        <div className="flex-1 flex flex-col relative">
            <div className="absolute top-0 left-0 z-10 p-2 flex space-x-2 pointer-events-none">
                <div className="bg-[#1e222d] shadow px-2 py-1 rounded border border-gray-600 text-xs font-bold text-white">
                    {activeSymbolObj.display} <span className="text-gray-400 font-normal">{timeframe}</span>
                </div>
            </div>

            {prices[activeSymbol] && (
                <OrderPanel 
                    symbolObj={activeSymbolObj} 
                    price={prices[activeSymbol]} 
                    onTrade={handleTrade} 
                />
            )}

            <Chart 
                data={chartData} 
                activeSymbolObj={activeSymbolObj} 
                currentPrice={prices[activeSymbol]?.close || 0}
                timeframe={timeframe}
            />
        </div>
      </div>

      <Terminal 
        positions={positions} 
        balance={balance} 
        equity={equity} 
        currentPrices={prices} 
        onClosePosition={closePosition}
        activeSymbolObj={activeSymbolObj}
      />
    </div>
  );
}
